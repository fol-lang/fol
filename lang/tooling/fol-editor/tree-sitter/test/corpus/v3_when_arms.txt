fun[] classify(value: int, maybe: opt[int]): int = {
    var probe: int = value;
    when(probe) {
        is(1) { return 1; }
        in(2..4) { return 2; }
        has(5) { return 5; }
        of(int) { return 6; }
        case(true) { return 8; }
        * { return 0; }
    }
    when(maybe) {
        on(present) { return present; }
        * { return 0; }
    }
};
