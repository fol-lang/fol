typ Counter: rec = {
    value: int,
};

fun[] worker(counter[mux]: Counter): int = {
    counter.lock();
    counter.value = counter.value + 1;
    counter.unlock();
    return counter.value;
};

fun[] channels(first: chn[int], second: chn[int], counter[mux]: Counter): int = {
    [>]worker(counter);
    [>]fun()[first[tx]] = {
        1 | first[tx];
    };
    var[mut] result: int = 0;
    select {
        when first as value {
            result = value;
        }
        when second as value {
            result = value;
        }
        * {
            result = result;
        }
    };
    return result;
};
