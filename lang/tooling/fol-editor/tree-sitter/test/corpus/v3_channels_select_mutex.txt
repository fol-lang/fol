typ Counter: rec = {
    value: int,
};

fun[] worker(counter: mux[Counter]): int = {
    var[mut, bor] guard: Counter = ([bor]counter).lock();
    guard.value = guard.value + 1;
    return guard.value;
};

fun[] channels(first: chn[int], second: chn[int], counter: mux[Counter]): int = {
    [>]worker(counter);
    [>]fun()[first[tx]] = {
        var sent: err[int] = 1 | first[tx];
    };
    var[mut] result: int = 0;
    select {
        when first as value {
            result = value;
        }
        when second as value {
            result = value;
        }
        * {
            result = result;
        }
    };
    return result;
};
