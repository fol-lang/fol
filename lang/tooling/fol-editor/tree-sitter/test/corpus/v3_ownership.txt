typ Parcel: rec = {
    value: int,
};

fun[] ownership(owner: Parcel): int = {
    @var heap: Parcel = { value = 1 };
    ~var replacement: Parcel = { value = 2 };
    var[bor] view: Parcel = owner;
    var[mut] mutable_owner: Parcel = replacement;
    var[mut, bor] mutable_view: Parcel = mutable_owner;
    mutable_view.value = heap.value;
    !mutable_view;
    !view;
    return mutable_owner.value;
};
