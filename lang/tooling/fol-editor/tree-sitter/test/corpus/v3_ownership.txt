typ Parcel: rec = {
    value: int,
};

fun[] inspect(parcel[bor]: Parcel): int = {
    return parcel.value;
};

fun[] main(): int = {
    var owner: Parcel = { value = 7 };
    var[bor] view: Parcel = owner;
    var first: int = inspect(view);
    var second: int = inspect(view);
    [end]view;
    return first + second;
};
